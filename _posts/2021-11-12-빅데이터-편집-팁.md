## 중복 검사를 빠르게 하기 위해 문자열로 변환

row = []
for idx, data in enumerate(all_data):
    row.append(str(data))

print(len(row))
print(len(set(row)))
print(f"중복건수 : {len(row) - len(set(row))}")
